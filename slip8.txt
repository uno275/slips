Slip 8 – Program 1
3 threads to print text n times


class TextThread extends Thread {
    String msg;
    int n;

    TextThread(String msg, int n) {
        this.msg = msg;
        this.n = n;
    }

    public void run() {
        for (int i = 1; i <= n; i++) {
            System.out.println(msg);
        }
    }
}

public class ThreadTextDemo {
    public static void main(String[] args) {
        TextThread t1 = new TextThread("COVID19", 10);
        TextThread t2 = new TextThread("LOCKDOWN2020", 20);
        TextThread t3 = new TextThread("VACCINATED2021", 30);

        t1.start();
        t2.start();
        t3.start();
    }
}




javac ThreadTextDemo.java
java ThreadTextDemo

<%
    int n = Integer.parseInt(request.getParameter("num"));
    int i, count = 0;

    for(i = 1; i <= n; i++) {
        if(n % i == 0) {
            count++;
        }
    }

    String result;
    if(count == 2)
        result = n + " is Prime Number";
