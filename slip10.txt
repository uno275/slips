Q1) MPI Program – Sum and Average of 1000 Numbers


#include <mpi.h>
#include <stdio.h>
#include <stdlib.h>
#include <time.h>

#define SIZE 1000

int main(int argc, char *argv[])
{
    int rank, size, i;
    int data[SIZE];
    int local_n;
    int local_sum = 0, total_sum = 0;
    double average;
    int *sub_data;

    MPI_Init(&argc, &argv);
    MPI_Comm_rank(MPI_COMM_WORLD, &rank);
    MPI_Comm_size(MPI_COMM_WORLD, &size);

    local_n = SIZE / size;
    sub_data = (int*)malloc(local_n * sizeof(int));

    if(rank == 0)
    {
        srand(time(NULL));
        for(i = 0; i < SIZE; i++)
            data[i] = rand() % 100;

        printf("Generated 1000 random numbers\n");
    }

    MPI_Scatter(data, local_n, MPI_INT, sub_data, local_n, MPI_INT, 0, MPI_COMM_WORLD);

    for(i = 0; i < local_n; i++)
        local_sum += sub_data[i];

    printf("Process %d local sum = %d\n", rank, local_sum);

    MPI_Reduce(&local_sum, &total_sum, 1, MPI_INT, MPI_SUM, 0, MPI_COMM_WORLD);

    if(rank == 0)
    {
        average = (double)total_sum / SIZE;
        printf("\nTotal Sum = %d\n", total_sum);
        printf("Average = %.2f\n", average);
    }

    free(sub_data);
    MPI_Finalize();
    return 0;
}






Q.2 Write a simulation program for disk scheduling using C-SCAN algorithm. Accept
total number of disk blocks, disk request string, and current head position from the user.
Display the list of request in the order in which it is served. Also display the total
number of head moments.
 33, 99, 142, 52, 197, 79, 46, 65
Start Head Position: 72
Direction: Left




#include <stdio.h>
#include <stdlib.h>

void sort(int arr[], int n)
{
    int i,j,temp;
    for(i=0;i<n-1;i++)
        for(j=0;j<n-i-1;j++)
            if(arr[j] > arr[j+1])
            {
                temp=arr[j];
                arr[j]=arr[j+1];
                arr[j+1]=temp;
            }
}

int main()
{
    int req[100], left[100], right[100];
    int n, head, disk_size, i;
    int lcount=0, rcount=0, total=0;

    printf("Enter total number of requests: ");
    scanf("%d",&n);

    printf("Enter request sequence:\n");
    for(i=0;i<n;i++)
        scanf("%d",&req[i]);

    printf("Enter initial head position: ");
    scanf("%d",&head);

    printf("Enter disk size: ");
    scanf("%d",&disk_size);

    for(i=0;i<n;i++)
    {
        if(req[i] < head)
            left[lcount++] = req[i];
        else
            right[rcount++] = req[i];
    }

    left[lcount++] = 0;
    right[rcount++] = disk_size-1;

    sort(left,lcount);
    sort(right,rcount);

    printf("\nOrder of service using C-SCAN:\n");
    printf("%d ",head);

    for(i=lcount-1;i>=0;i--)
    {
        total += abs(head-left[i]);
        head = left[i];
        printf("-> %d ",head);
    }

    for(i=rcount-1;i>=0;i--)
    {
        total += abs(head-right[i]);
        head = right[i];
        printf("-> %d ",head);
    }

    printf("\n\nTotal head movements = %d\n",total);

    return 0;
}
