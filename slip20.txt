Q.1 Write a simulation program for disk scheduling using SCAN algorithm. Accept total
number of disk blocks, disk request string, and current head position from the user.
Display the list of request in the order in which it is served. Also display the total number
of head moments.
33, 99, 142, 52, 197, 79, 46, 65
Start Head Position: 72
 Direction: User defined





#include <stdio.h>
#include <stdlib.h>

void sort(int arr[], int n)
{
    int i, j, temp;
    for(i = 0; i < n - 1; i++)
    {
        for(j = 0; j < n - i - 1; j++)
        {
            if(arr[j] > arr[j + 1])
            {
                temp = arr[j];
                arr[j] = arr[j + 1];
                arr[j + 1] = temp;
            }
        }
    }
}

int main()
{
    int req[100], left[100], right[100];
    int n, head, disk_size, i, direction;
    int lcount = 0, rcount = 0, total = 0;

    printf("Enter total number of requests: ");
    scanf("%d", &n);

    printf("Enter request sequence:\n");
    for(i = 0; i < n; i++)
        scanf("%d", &req[i]);

    printf("Enter initial head position: ");
    scanf("%d", &head);

    printf("Enter disk size: ");
    scanf("%d", &disk_size);

    printf("Enter direction (0 for Left, 1 for Right): ");
    scanf("%d", &direction);

    for(i = 0; i < n; i++)
    {
        if(req[i] < head)
            left[lcount++] = req[i];
        else
            right[rcount++] = req[i];
    }

    sort(left, lcount);
    sort(right, rcount);

    printf("\nOrder of service using SCAN:\n");
    printf("%d ", head);

    if(direction == 0)
    {
        left[lcount++] = 0;
        sort(left, lcount);

        for(i = lcount - 1; i >= 0; i--)
        {
            total += abs(head - left[i]);
            head = left[i];
            printf("-> %d ", head);
        }

        for(i = 0; i < rcount; i++)
        {
            total += abs(head - right[i]);
            head = right[i];
            printf("-> %d ", head);
        }
    }
    else
    {
        right[rcount++] = disk_size - 1;
        sort(right, rcount);

        for(i = 0; i < rcount; i++)
        {
            total += abs(head - right[i]);
            head = right[i];
            printf("-> %d ", head);
        }

        for(i = lcount - 1; i >= 0; i--)
        {
            total += abs(head - left[i]);
            head = left[i];
            printf("-> %d ", head);
        }
    }

    printf("\n\nTotal head movements = %d\n", total);

    return 0;
}

Q.2 Write an MPI program to find the max number from randomly generated 1000 numbers
(stored in array) on a cluster (Hint: Use MPI_Reduce)




#include <mpi.h>
#include <stdio.h>
#include <stdlib.h>
#include <time.h>

#define SIZE 1000

int main(int argc, char *argv[])
{
    int rank, nprocs, i;
    int data[SIZE];
    int local_n;
    int local_max, global_max;
    int *sub_data;

    MPI_Init(&argc, &argv);
    MPI_Comm_rank(MPI_COMM_WORLD, &rank);
    MPI_Comm_size(MPI_COMM_WORLD, &nprocs);

    if (SIZE % nprocs != 0)
    {
        if (rank == 0)
            printf("1000 numbers cannot be divided equally among %d processes\n", nprocs);
        MPI_Finalize();
        return 0;
    }

    local_n = SIZE / nprocs;
    sub_data = (int *)malloc(local_n * sizeof(int));

    if (rank == 0)
    {
        srand(time(NULL));
        for (i = 0; i < SIZE; i++)
            data[i] = rand() % 1000;

        printf("Generated 1000 random numbers\n");
    }

    MPI_Scatter(data, local_n, MPI_INT, sub_data, local_n, MPI_INT, 0, MPI_COMM_WORLD);

    local_max = sub_data[0];
    for (i = 1; i < local_n; i++)
    {
        if (sub_data[i] > local_max)
            local_max = sub_data[i];
    }

    printf("Process %d local max = %d\n", rank, local_max);

    MPI_Reduce(&local_max, &global_max, 1, MPI_INT, MPI_MAX, 0, MPI_COMM_WORLD);

    if (rank == 0)
    {
        printf("Maximum number = %d\n", global_max);
    }

    free(sub_data);
    MPI_Finalize();
    return 0;
}
