Q.1 Write an MPI program to calculate sum of all odd randomly generated 1000 numbers
(stored in array) on a cluster. 



#include <mpi.h>
#include <stdio.h>
#include <stdlib.h>
#include <time.h>

#define SIZE 1000

int main(int argc, char *argv[])
{
    int rank, nprocs, i;
    int data[SIZE];
    int local_n, local_odd_sum = 0, total_odd_sum = 0;
    int *sub_data;

    MPI_Init(&argc, &argv);
    MPI_Comm_rank(MPI_COMM_WORLD, &rank);
    MPI_Comm_size(MPI_COMM_WORLD, &nprocs);

    if (SIZE % nprocs != 0)
    {
        if (rank == 0)
            printf("1000 numbers cannot be divided equally among %d processes\n", nprocs);
        MPI_Finalize();
        return 0;
    }

    local_n = SIZE / nprocs;
    sub_data = (int *)malloc(local_n * sizeof(int));

    if (rank == 0)
    {
        srand(time(NULL));
        for (i = 0; i < SIZE; i++)
            data[i] = rand() % 1000;

        printf("Generated 1000 random numbers\n");
    }

    MPI_Scatter(data, local_n, MPI_INT, sub_data, local_n, MPI_INT, 0, MPI_COMM_WORLD);

    for (i = 0; i < local_n; i++)
    {
        if (sub_data[i] % 2 != 0)
            local_odd_sum += sub_data[i];
    }

    printf("Process %d local odd sum = %d\n", rank, local_odd_sum);

    MPI_Reduce(&local_odd_sum, &total_odd_sum, 1, MPI_INT, MPI_SUM, 0, MPI_COMM_WORLD);

    if (rank == 0)
    {
        printf("Total sum of odd numbers = %d\n", total_odd_sum);
    }

    free(sub_data);
    MPI_Finalize();
    return 0;
}









Q2) Sequential (Contiguous) File Allocation Method

Menu:

Show Bit Vector
Delete already created file
Exit




#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <time.h>

#define MAX 100

struct file
{
    char name[20];
    int start;
    int length;
};

int bit[MAX];
int n, fileCount = 0;
struct file dir[20];

void initDisk()
{
    int i;
    srand(time(0));
    for(i = 0; i < n; i++)
        bit[i] = rand() % 2;
}

void createSampleFiles()
{
    int i, j;

    for(i = 0; i < n - 2 && fileCount < 2; i++)
    {
        if(bit[i] == 0 && bit[i + 1] == 0)
        {
            sprintf(dir[fileCount].name, "file%d", fileCount + 1);
            dir[fileCount].start = i;
            dir[fileCount].length = 2;

            for(j = i; j < i + 2; j++)
                bit[j] = 1;

            fileCount++;
            i++;
        }
    }
}

void showBitVector()
{
    int i;
    printf("\nBit Vector:\n");
    for(i = 0; i < n; i++)
        printf("%d ", bit[i]);
    printf("\n");
}

void deleteFile()
{
    char fname[20];
    int i, j, found = 0;

    printf("Enter file name to delete: ");
    scanf("%s", fname);

    for(i = 0; i < fileCount; i++)
    {
        if(strcmp(dir[i].name, fname) == 0)
        {
            found = 1;

            for(j = dir[i].start; j < dir[i].start + dir[i].length; j++)
                bit[j] = 0;

            for(j = i; j < fileCount - 1; j++)
                dir[j] = dir[j + 1];

            fileCount--;
            printf("File deleted successfully.\n");
            break;
        }
    }

    if(!found)
        printf("File not found.\n");
}

int main()
{
    int choice;

    printf("Enter total number of disk blocks: ");
    scanf("%d", &n);

    initDisk();
    createSampleFiles();

    do
    {
        printf("\n--- MENU ---\n");
        printf("1. Show Bit Vector\n");
        printf("2. Delete already created file\n");
        printf("3. Exit\n");
        printf("Enter choice: ");
        scanf("%d", &choice);

        switch(choice)
        {
            case 1:
                showBitVector();
                break;
            case 2:
                deleteFile();
                break;
            case 3:
                printf("Exiting...\n");
                break;
            default:
                printf("Invalid choice.\n");
        }
    } while(choice != 3);

    return 0;
}
