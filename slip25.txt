2. Write a Java Program for the following: Assume database is already created. 







Save as DDLQuery.java



import java.awt.*;
import java.awt.event.*;
import java.sql.*;

public class DDLQuery extends Frame implements ActionListener {

    TextField tf;
    Button b1,b2,b3;

    Connection con;
    Statement st;

    DDLQuery()
    {
        Label l = new Label("Type Your DDL Query Here");

        tf = new TextField(40);

        b1 = new Button("Create Table");
        b2 = new Button("Alter Table");
        b3 = new Button("Drop Table");

        setLayout(new FlowLayout());

        add(l);
        add(tf);
        add(b1);
        add(b2);
        add(b3);

        b1.addActionListener(this);
        b2.addActionListener(this);
        b3.addActionListener(this);

        setSize(400,200);
        setVisible(true);

        try
        {
            Class.forName("org.postgresql.Driver");

            con = DriverManager.getConnection(
            "jdbc:postgresql://localhost:5432/testdb",
            "postgres",
            "password");

            st = con.createStatement();
        }
        catch(Exception e)
        {
            System.out.println(e);
        }
    }

    public void actionPerformed(ActionEvent e)
    {
        try
        {
            String q = tf.getText();

            st.executeUpdate(q);

            System.out.println("Query Executed Successfully");

        }
        catch(Exception ex)
        {
            System.out.println(ex);
        }
    }

    public static void main(String args[])
    {
        new DDLQuery();
    }
}




Example Queries
Create Table
CREATE TABLE student(id INT, name VARCHAR(50));

Alter Table
ALTER TABLE student ADD age INT;

Drop Table
DROP TABLE student;

Output

A window will appear with:

Type Your DDL Query Here

[ TextField ]

[Create Table] [Alter Table] [Drop Table]
