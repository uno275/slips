1.Write a Java program to display all the alphabets between ‘A’ to ‘Z’ after every 2 seconds.

class AlphabetDelay
{
    public static void main(String args[])
    {
        try
        {
            for(char ch='A'; ch<='Z'; ch++)
            {
                System.out.println(ch);
                Thread.sleep(2000);   // 2000 milliseconds = 2 seconds
            }
        }
        catch(Exception e)
        {
            System.out.println(e);
        }
    }
}


javac AlphabetDelay.java
java AlphabetDelay



2. Write a Java program to accept the details of Employee (Eno, EName, Designation, Salary) from a user and store it into the database. (Use Swing)

Open terminal and run: psql

CREATE TABLE Employee (
    Eno INT PRIMARY KEY,
    EName VARCHAR(50),
    Designation VARCHAR(50),
    Salary DOUBLE PRECISION
);



import javax.swing.*;
import java.awt.*;
import java.awt.event.*;
import java.sql.*;

public class EmployeeForm extends JFrame implements ActionListener {

    JLabel l1, l2, l3, l4;
    JTextField t1, t2, t3, t4;
    JButton b1, b2;

    Connection con;
    PreparedStatement pst;

    public EmployeeForm() {
        setTitle("Employee Form");
        setSize(400, 300);
        setLayout(new GridLayout(5, 2, 10, 10));
        setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);

        l1 = new JLabel("Employee No:");
        l2 = new JLabel("Employee Name:");
        l3 = new JLabel("Designation:");
        l4 = new JLabel("Salary:");

        t1 = new JTextField();
        t2 = new JTextField();
        t3 = new JTextField();
        t4 = new JTextField();

        b1 = new JButton("Save");
        b2 = new JButton("Clear");

        add(l1); add(t1);
        add(l2); add(t2);
        add(l3); add(t3);
        add(l4); add(t4);
        add(b1); add(b2);

        b1.addActionListener(this);
        b2.addActionListener(this);

        connectDB();

        setVisible(true);
    }

    public void connectDB() {
        try {
            Class.forName("org.postgresql.Driver");
            con = DriverManager.getConnection(
                "jdbc:postgresql://localhost:5432/postgres",
                "postgres",
                "postgres"
            );
            System.out.println("Database Connected Successfully");
        } catch (Exception e) {
            JOptionPane.showMessageDialog(this, "DB Connection Error: " + e.getMessage());
        }
    }

    public void actionPerformed(ActionEvent ae) {
        if (ae.getSource() == b1) {
            try {
                int eno = Integer.parseInt(t1.getText());
                String ename = t2.getText();
                String desig = t3.getText();
                double salary = Double.parseDouble(t4.getText());

                pst = con.prepareStatement("INSERT INTO Employee VALUES (?, ?, ?, ?)");
                pst.setInt(1, eno);
                pst.setString(2, ename);
                pst.setString(3, desig);
                pst.setDouble(4, salary);

                int result = pst.executeUpdate();

                if (result > 0) {
                    JOptionPane.showMessageDialog(this, "Record Inserted Successfully");
                } else {
                    JOptionPane.showMessageDialog(this, "Record Not Inserted");
                }

            } catch (Exception e) {
                JOptionPane.showMessageDialog(this, "Error: " + e.getMessage());
            }
        }

        if (ae.getSource() == b2) {
            t1.setText("");
            t2.setText("");
            t3.setText("");
            t4.setText("");
        }
    }

    public static void main(String[] args) {
        new EmployeeForm();
    }
}
