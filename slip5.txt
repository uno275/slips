Q.1 Consider a system with ‘m’ processes and ‘n’ resource types. Accept number of
instances for every resource type. For each process accept the allocation and maximum
requirement matrices. Write a program to display the contents of need matrix and to
check if the given request of a process can be granted immediately or not


#include <stdio.h>

#define MAXP 10
#define MAXR 10

int main() {
    int p, r;
    int allocation[MAXP][MAXR], max[MAXP][MAXR], need[MAXP][MAXR];
    int total[MAXR], allocated[MAXR] = {0}, available[MAXR];
    int request[MAXR], process;
    int i, j, flag = 1;

    printf("Enter number of processes: ");
    scanf("%d", &p);

    printf("Enter number of resource types: ");
    scanf("%d", &r);

    printf("Enter number of instances for each resource:\n");
    for(i = 0; i < r; i++) {
        printf("Resource %d: ", i);
        scanf("%d", &total[i]);
    }

    printf("Enter Allocation Matrix:\n");
    for(i = 0; i < p; i++) {
        for(j = 0; j < r; j++) {
            scanf("%d", &allocation[i][j]);
            allocated[j] += allocation[i][j];
        }
    }

    printf("Enter Maximum Matrix:\n");
    for(i = 0; i < p; i++) {
        for(j = 0; j < r; j++) {
            scanf("%d", &max[i][j]);
        }
    }

    for(i = 0; i < r; i++) {
        available[i] = total[i] - allocated[i];
    }

    for(i = 0; i < p; i++) {
        for(j = 0; j < r; j++) {
            need[i][j] = max[i][j] - allocation[i][j];
        }
    }

    printf("\nNeed Matrix:\n");
    for(i = 0; i < p; i++) {
        printf("P%d: ", i);
        for(j = 0; j < r; j++) {
            printf("%d ", need[i][j]);
        }
        printf("\n");
    }

    printf("\nAvailable Resources:\n");
    for(i = 0; i < r; i++) {
        printf("%d ", available[i]);
    }
    printf("\n");

    printf("\nEnter process number for request: ");
    scanf("%d", &process);

    printf("Enter request for process P%d:\n", process);
    for(i = 0; i < r; i++) {
        scanf("%d", &request[i]);
    }

    for(i = 0; i < r; i++) {
        if(request[i] > need[process][i]) {
            flag = 0;
            printf("\nRequest cannot be granted immediately.\n");
            printf("Reason: Request is greater than Need.\n");
            break;
        }
    }

    if(flag == 1) {
        for(i = 0; i < r; i++) {
            if(request[i] > available[i]) {
                flag = 0;
                printf("\nRequest cannot be granted immediately.\n");
                printf("Reason: Resources are not available.\n");
                break;
            }
        }
    }

    if(flag == 1) {
        printf("\nRequest can be granted immediately.\n");
    }

    return 0;
}

Q.2 Write an MPI program to find the max number from randomly generated 1000 numbers
(stored in array) on a cluster (Hint: Use MPI_Reduce)



#include <mpi.h>
#include <stdio.h>
#include <stdlib.h>
#include <time.h>

#define SIZE 1000

int main(int argc, char *argv[]) {
    int rank, nprocs, i;
    int data[SIZE];
    int local_n;
    int local_max, global_max;
    int *sub_data;

    MPI_Init(&argc, &argv);
    MPI_Comm_rank(MPI_COMM_WORLD, &rank);
    MPI_Comm_size(MPI_COMM_WORLD, &nprocs);

    if (SIZE % nprocs != 0) {
        if (rank == 0)
            printf("1000 numbers cannot be divided equally among %d processes\n", nprocs);
        MPI_Finalize();
        return 0;
    }

    local_n = SIZE / nprocs;
    sub_data = (int *)malloc(local_n * sizeof(int));

    if (rank == 0) {
        srand(time(NULL));
        for (i = 0; i < SIZE; i++) {
            data[i] = rand() % 1000;
        }
        printf("Generated 1000 random numbers\n");
    }

    MPI_Scatter(data, local_n, MPI_INT, sub_data, local_n, MPI_INT, 0, MPI_COMM_WORLD);

    local_max = sub_data[0];
    for (i = 1; i < local_n; i++) {
        if (sub_data[i] > local_max)
            local_max = sub_data[i];
    }

    printf("Process %d local max = %d\n", rank, local_max);

    MPI_Reduce(&local_max, &global_max, 1, MPI_INT, MPI_MAX, 0, MPI_COMM_WORLD);

    if (rank == 0) {
        printf("Maximum number = %d\n", global_max);
    }

    free(sub_data);
    MPI_Finalize();
    return 0;
}
