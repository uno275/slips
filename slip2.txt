Q.1 Write a program to simulate Linked file allocation method. Assume disk with n
number of blocks. Give value of n as input. Randomly mark some block as allocated and
accordingly maintain the list of free blocks Write menu driver program with menu
options as mentioned below and implement each option.
• Show Bit Vector
• Create New File
• Show Directory
• Exit


#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <time.h>

#define MAX 100

struct file {
    char name[20];
    int start;
    int blocks[20];
    int count;
};

int bitVector[MAX];
int n, fileCount = 0;
struct file directory[20];

void initializeDisk() {
    int i;
    srand(time(0));

    for(i = 0; i < n; i++) {
        bitVector[i] = rand() % 2;   // 0 = free, 1 = allocated
    }
}

void showBitVector() {
    int i;
    printf("\nBit Vector:\n");
    for(i = 0; i < n; i++) {
        printf("%d ", bitVector[i]);
    }
    printf("\n");
}

void createFile() {
    char fname[20];
    int bcount, i, j, freeCount = 0;

    printf("Enter file name: ");
    scanf("%s", fname);

    printf("Enter number of blocks required: ");
    scanf("%d", &bcount);

    for(i = 0; i < n; i++) {
        if(bitVector[i] == 0)
            freeCount++;
    }

    if(freeCount < bcount) {
        printf("Not enough free blocks available.\n");
        return;
    }

    strcpy(directory[fileCount].name, fname);
    directory[fileCount].count = bcount;

    j = 0;
    for(i = 0; i < n && j < bcount; i++) {
        if(bitVector[i] == 0) {
            bitVector[i] = 1;
            directory[fileCount].blocks[j] = i;
            j++;
        }
    }

    directory[fileCount].start = directory[fileCount].blocks[0];
    fileCount++;

    printf("File created successfully using linked allocation.\n");
}

void showDirectory() {
    int i, j;

    if(fileCount == 0) {
        printf("\nDirectory is empty.\n");
        return;
    }

    printf("\nDirectory:\n");
    printf("File Name\tStart Block\tAllocated Blocks\n");

    for(i = 0; i < fileCount; i++) {
        printf("%s\t\t%d\t\t", directory[i].name, directory[i].start);
        for(j = 0; j < directory[i].count; j++) {
            printf("%d", directory[i].blocks[j]);
            if(j != directory[i].count - 1)
                printf(" -> ");
        }
        printf("\n");
    }
}

int main() {
    int choice;

    printf("Enter total number of disk blocks: ");
    scanf("%d", &n);

    initializeDisk();

    do {
        printf("\n----- MENU -----\n");
        printf("1. Show Bit Vector\n");
        printf("2. Create New File\n");
        printf("3. Show Directory\n");
        printf("4. Exit\n");
        printf("Enter your choice: ");
        scanf("%d", &choice);

        switch(choice) {
            case 1:
                showBitVector();
                break;
            case 2:
                createFile();
                break;
            case 3:
                showDirectory();
                break;
            case 4:
                printf("Exiting...\n");
                break;
            default:
                printf("Invalid choice!\n");
        }
    } while(choice != 4);

    return 0;
}

Q.2 Write an MPI program to calculate sum of randomly generated 1000 numbers
(stored in array) on a cluster


#include <mpi.h>
#include <stdio.h>
#include <stdlib.h>
#include <time.h>

#define SIZE 1000

int main(int argc, char *argv[])
{
    int rank, nprocs, i;
    int local_n, local_sum = 0, total_sum = 0;
    int data[SIZE];
    int *sub_data;

    MPI_Init(&argc, &argv);
    MPI_Comm_rank(MPI_COMM_WORLD, &rank);
    MPI_Comm_size(MPI_COMM_WORLD, &nprocs);

    if (SIZE % nprocs != 0)
    {
        if (rank == 0)
            printf("1000 numbers cannot be divided equally among %d processes\n", nprocs);
        MPI_Finalize();
        return 0;
    }

    local_n = SIZE / nprocs;
    sub_data = (int *)malloc(local_n * sizeof(int));

    if (rank == 0)
    {
        srand(time(NULL));
        for (i = 0; i < SIZE; i++)
        {
            data[i] = rand() % 100;
        }
        printf("Generated 1000 random numbers\n");
    }

    MPI_Scatter(data, local_n, MPI_INT,
                sub_data, local_n, MPI_INT,
                0, MPI_COMM_WORLD);

    for (i = 0; i < local_n; i++)
    {
        local_sum += sub_data[i];
    }

    printf("Process %d local sum = %d\n", rank, local_sum);

    MPI_Reduce(&local_sum, &total_sum, 1, MPI_INT, MPI_SUM, 0, MPI_COMM_WORLD);

    if (rank == 0)
    {
        printf("Total sum = %d\n", total_sum);
    }

    free(sub_data);
    MPI_Finalize();
    return 0;
}
