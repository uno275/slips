Write a java program to read ‘N’ names of your friends, store it into HashSet and display them in ascending order.



import java.util.*;

public class FriendNames {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        HashSet<String> names = new HashSet<String>();

        System.out.print("Enter number of friends: ");
        int n = sc.nextInt();
        sc.nextLine();

        for (int i = 1; i <= n; i++) {
            System.out.print("Enter friend name " + i + ": ");
            String name = sc.nextLine();
            names.add(name);
        }

        TreeSet<String> sortedNames = new TreeSet<String>(names);

        System.out.println("\nNames in ascending order:");
        for (String s : sortedNames) {
            System.out.println(s);
        }
    }
}

Program 2
Design a servlet that provides information about a HTTP request from a client, such as IP-Address and browser type. The servlet also provides information about the server on which the servlet is running, such as the operating system type, and the names of currently loaded servlets.




import java.io.*;
import javax.servlet.*;
import javax.servlet.http.*;
import java.util.*;

public class InfoServlet extends HttpServlet {

    public void doGet(HttpServletRequest request, HttpServletResponse response)
            throws ServletException, IOException {

        response.setContentType("text/html");
        PrintWriter out = response.getWriter();

        String ipAddress = request.getRemoteAddr();
        String browser = request.getHeader("User-Agent");

        ServletContext context = getServletContext();
        Enumeration<String> servletNames = context.getServletNames();

        out.println("<html>");
        out.println("<head><title>Request and Server Info</title></head>");
        out.println("<body>");
        out.println("<h2>HTTP Request Information</h2>");
        out.println("<p><b>Client IP Address:</b> " + ipAddress + "</p>");
        out.println("<p><b>Browser Type:</b> " + browser + "</p>");

        out.println("<h2>Server Information</h2>");
        out.println("<p><b>Operating System:</b> " + System.getProperty("os.name") + "</p>");

        out.println("<h3>Currently Loaded Servlets</h3>");
        out.println("<ul>");
        while (servletNames.hasMoreElements()) {
            out.println("<li>" + servletNames.nextElement() + "</li>");
        }
        out.println("</ul>");

        out.println("</body>");
        out.println("</html>");
    }
}





web.xml mapping


<web-app>
    <servlet>
        <servlet-name>InfoServlet</servlet-name>
        <servlet-class>InfoServlet</servlet-class>
    </servlet>

    <servlet-mapping>
        <servlet-name>InfoServlet</servlet-name>
        <url-pattern>/info</url-pattern>
    </servlet-mapping>
</web-app>
