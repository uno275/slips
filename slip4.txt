Program 1
Write a Java program using Runnable interface to blink Text on the frame.


import javax.swing.*;
import java.awt.*;

public class BlinkText extends JFrame implements Runnable
{
    JLabel l;
    Thread t;
    boolean visible = true;

    BlinkText()
    {
        setTitle("Blinking Text");
        setSize(400, 200);
        setLayout(new FlowLayout());
        setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);

        l = new JLabel("WELCOME");
        l.setFont(new Font("Arial", Font.BOLD, 30));
        add(l);

        setVisible(true);

        t = new Thread(this);
        t.start();
    }

    public void run()
    {
        try
        {
            while(true)
            {
                l.setVisible(visible);
                visible = !visible;
                Thread.sleep(500);
            }
        }
        catch(Exception e)
        {
            System.out.println(e);
        }
    }

    public static void main(String args[])
    {
        new BlinkText();
    }
}




Compile:

javac BlinkText.java
java BlinkText


Output
A frame opens on the screen.
Text "WELCOME" blinks continuously.
Screen View
--------------------------
|                        |
|       WELCOME          |
|                        |
--------------------------






Program 2
Write a Java program to store city names and their STD codes using an appropriate collection and perform:
Add new city and code
Remove city
Search city and display code





import java.util.*;

public class CitySTD
{
    public static void main(String args[])
    {
        Scanner sc = new Scanner(System.in);
        HashMap<String, String> hm = new HashMap<String, String>();

        int ch;
        String city, code;

        do
        {
            System.out.println("\n1. Add City");
            System.out.println("2. Remove City");
            System.out.println("3. Search City");
            System.out.println("4. Display All");
            System.out.println("5. Exit");
            System.out.print("Enter your choice: ");
            ch = sc.nextInt();
            sc.nextLine();

            switch(ch)
            {
                case 1:
                    System.out.print("Enter city name: ");
                    city = sc.nextLine();

                    if(hm.containsKey(city))
                    {
                        System.out.println("City already exists. No duplicates allowed.");
                    }
                    else
                    {
                        System.out.print("Enter STD code: ");
                        code = sc.nextLine();
                        hm.put(city, code);
                        System.out.println("City added successfully.");
                    }
                    break;

                case 2:
                    System.out.print("Enter city name to remove: ");
                    city = sc.nextLine();

                    if(hm.containsKey(city))
                    {
                        hm.remove(city);
                        System.out.println("City removed successfully.");
                    }
                    else
                    {
                        System.out.println("City not found.");
                    }
                    break;

                case 3:
                    System.out.print("Enter city name to search: ");
                    city = sc.nextLine();

                    if(hm.containsKey(city))
                    {
                        System.out.println("STD Code of " + city + " is: " + hm.get(city));
                    }
                    else
                    {
                        System.out.println("City not found.");
                    }
                    break;

                case 4:
                    System.out.println("Cities and STD Codes:");
                    for(Map.Entry<String, String> e : hm.entrySet())
                    {
                        System.out.println(e.getKey() + " -> " + e.getValue());
                    }
                    break;

                case 5:
                    System.out.println("Exiting...");
                    break;

                default:
                    System.out.println("Invalid choice.");
            }

        } while(ch != 5);
    }
}





Compile:


javac CitySTD.java
java CitySTD
