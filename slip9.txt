Slip 9 – Program 1
Write a Java program to create a thread for moving a ball inside a panel vertically. The ball should be created when the user clicks on the start button.



import javax.swing.*;
import java.awt.*;
import java.awt.event.*;

class BallPanel extends JPanel implements Runnable {
    int y = 20;
    Thread t;

    public void startBall() {
        t = new Thread(this);
        t.start();
    }

    public void run() {
        try {
            while (true) {
                y += 10;
                if (y > getHeight() - 30) {
                    y = 20;
                }
                repaint();
                Thread.sleep(200);
            }
        } catch (Exception e) {
            System.out.println(e);
        }
    }

    public void paintComponent(Graphics g) {
        super.paintComponent(g);
        g.setColor(Color.RED);
        g.fillOval(100, y, 30, 30);
    }
}

public class MovingBall extends JFrame implements ActionListener {
    JButton b;
    BallPanel p;

    MovingBall() {
        setTitle("Moving Ball");
        setSize(300, 400);
        setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);

        b = new JButton("Start");
        p = new BallPanel();

        add(b, BorderLayout.SOUTH);
        add(p, BorderLayout.CENTER);

        b.addActionListener(this);

        setVisible(true);
    }

    public void actionPerformed(ActionEvent e) {
        p.startBall();
    }

    public static void main(String args[]) {
        new MovingBall();
    }
}
