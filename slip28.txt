Q1) C-LOOK Disk Scheduling




#include <stdio.h>
#include <stdlib.h>

void sort(int arr[], int n)
{
    int i, j, temp;
    for(i = 0; i < n - 1; i++)
    {
        for(j = 0; j < n - i - 1; j++)
        {
            if(arr[j] > arr[j + 1])
            {
                temp = arr[j];
                arr[j] = arr[j + 1];
                arr[j + 1] = temp;
            }
        }
    }
}

int main()
{
    int req[100], left[100], right[100];
    int n, head, i, direction;
    int lcount = 0, rcount = 0, total = 0;

    printf("Enter total number of requests: ");
    scanf("%d", &n);

    printf("Enter request sequence:\n");
    for(i = 0; i < n; i++)
        scanf("%d", &req[i]);

    printf("Enter initial head position: ");
    scanf("%d", &head);

    printf("Enter direction (0 for Left, 1 for Right): ");
    scanf("%d", &direction);

    for(i = 0; i < n; i++)
    {
        if(req[i] < head)
            left[lcount++] = req[i];
        else
            right[rcount++] = req[i];
    }

    sort(left, lcount);
    sort(right, rcount);

    printf("\nOrder of service using C-LOOK:\n");
    printf("%d ", head);

    if(direction == 1)
    {
        for(i = 0; i < rcount; i++)
        {
            total += abs(head - right[i]);
            head = right[i];
            printf("-> %d ", head);
        }

        for(i = 0; i < lcount; i++)
        {
            total += abs(head - left[i]);
            head = left[i];
            printf("-> %d ", head);
        }
    }
    else
    {
        for(i = lcount - 1; i >= 0; i--)
        {
            total += abs(head - left[i]);
            head = left[i];
            printf("-> %d ", head);
        }

        for(i = rcount - 1; i >= 0; i--)
        {
            total += abs(head - right[i]);
            head = right[i];
            printf("-> %d ", head);
        }
    }

    printf("\n\nTotal head movements = %d\n", total);

    return 0;
}
