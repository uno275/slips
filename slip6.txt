Q.1 Write a program to simulate Linked file allocation method. Assume disk with n
number of blocks. Give value of n as input. Randomly mark some block as allocated and
accordingly maintain the list of free blocks Write menu driver program with menu
options as mentioned below and implement each option.
• Show Bit Vector
• Create New File
• Show Directory
• Exit


#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <time.h>

#define MAX 100

struct file {
    char name[20];
    int start;
    int blocks[20];
    int count;
};

int bit[MAX];
int n, fileCount = 0;
struct file dir[20];

void initDisk() {
    int i;
    srand(time(0));
    for(i = 0; i < n; i++) {
        bit[i] = rand() % 2;
    }
}

void showBitVector() {
    int i;
    printf("\nBit Vector:\n");
    for(i = 0; i < n; i++) {
        printf("%d ", bit[i]);
    }
    printf("\n");
}

void createFile() {
    char fname[20];
    int bcount, i, j = 0, freeCount = 0;

    printf("Enter file name: ");
    scanf("%s", fname);

    printf("Enter number of blocks required: ");
    scanf("%d", &bcount);

    for(i = 0; i < n; i++) {
        if(bit[i] == 0)
            freeCount++;
    }

    if(freeCount < bcount) {
        printf("Not enough free blocks.\n");
        return;
    }

    strcpy(dir[fileCount].name, fname);
    dir[fileCount].count = bcount;

    for(i = 0; i < n && j < bcount; i++) {
        if(bit[i] == 0) {
            bit[i] = 1;
            dir[fileCount].blocks[j] = i;
            j++;
        }
    }

    dir[fileCount].start = dir[fileCount].blocks[0];
    fileCount++;

    printf("File created successfully.\n");
}

void showDirectory() {
    int i, j;
    if(fileCount == 0) {
        printf("\nDirectory empty.\n");
        return;
    }

    printf("\nDirectory:\n");
    printf("File Name\tStart Block\tAllocated Blocks\n");
    for(i = 0; i < fileCount; i++) {
        printf("%s\t\t%d\t\t", dir[i].name, dir[i].start);
        for(j = 0; j < dir[i].count; j++) {
            printf("%d", dir[i].blocks[j]);
            if(j != dir[i].count - 1)
                printf(" -> ");
        }
        printf("\n");
    }
}

int main() {
    int choice;

    printf("Enter total number of disk blocks: ");
    scanf("%d", &n);

    initDisk();

    do {
        printf("\n--- MENU ---\n");
        printf("1. Show Bit Vector\n");
        printf("2. Create New File\n");
        printf("3. Show Directory\n");
        printf("4. Exit\n");
        printf("Enter choice: ");
        scanf("%d", &choice);

        switch(choice) {
            case 1:
                showBitVector();
                break;
            case 2:
                createFile();
                break;
            case 3:
                showDirectory();
                break;
            case 4:
                printf("Exiting...\n");
                break;
            default:
                printf("Invalid choice.\n");
        }
    } while(choice != 4);

    return 0;
}

Q.2 Write a simulation program for disk scheduling using C-SCAN algorithm. Accept total
number of disk blocks, disk request string, and current head position from the user. Display
the list of request in the order in which it is served. Also display the total number of head
moments..
80, 150, 60,135, 40, 35, 170
Starting Head Position: 70
Direction: Right




#include <stdio.h>
#include <stdlib.h>

void sort(int arr[], int n) {
    int i, j, temp;
    for(i = 0; i < n - 1; i++) {
        for(j = 0; j < n - i - 1; j++) {
            if(arr[j] > arr[j + 1]) {
                temp = arr[j];
                arr[j] = arr[j + 1];
                arr[j + 1] = temp;
            }
        }
    }
}

int main() {
    int req[100], left[100], right[100];
    int n, head, disk_size, i;
    int lcount = 0, rcount = 0, total = 0;

    printf("Enter total number of requests: ");
    scanf("%d", &n);

    printf("Enter request sequence:\n");
    for(i = 0; i < n; i++) {
        scanf("%d", &req[i]);
    }

    printf("Enter initial head position: ");
    scanf("%d", &head);

    printf("Enter disk size: ");
    scanf("%d", &disk_size);

    for(i = 0; i < n; i++) {
        if(req[i] < head)
            left[lcount++] = req[i];
        else
            right[rcount++] = req[i];
    }

    right[rcount++] = disk_size - 1;
    left[lcount++] = 0;

    sort(left, lcount);
    sort(right, rcount);

    printf("\nOrder of service using C-SCAN:\n");
    printf("%d ", head);

    for(i = 0; i < rcount; i++) {
        total += abs(head - right[i]);
        head = right[i];
        printf("-> %d ", head);
    }

    head = 0;
    printf("-> %d ", head);
    total += (disk_size - 1);

    for(i = 0; i < lcount - 1; i++) {
        total += abs(head - left[i]);
        head = left[i];
        printf("-> %d ", head);
    }

    printf("\n\nTotal head movements = %d\n", total);

    return 0;
}
