1. Write a java program for the implementation of synchronization




Save as SyncDemo.java

class Table {
    synchronized void printTable(int n) {
        for (int i = 1; i <= 5; i++) {
            System.out.println(n + " x " + i + " = " + (n * i));
            try {
                Thread.sleep(500);
            } catch (Exception e) {
                System.out.println(e);
            }
        }
    }
}

class MyThread1 extends Thread {
    Table t;

    MyThread1(Table t) {
        this.t = t;
    }

    public void run() {
        t.printTable(5);
    }
}

class MyThread2 extends Thread {
    Table t;

    MyThread2(Table t) {
        this.t = t;
    }

    public void run() {
        t.printTable(10);
    }
}

public class SyncDemo {
    public static void main(String[] args) {
        Table obj = new Table();

        MyThread1 t1 = new MyThread1(obj);
        MyThread2 t2 = new MyThread2(obj);

        t1.start();
        t2.start();
    }
}




Compile:
javac SyncDemo.java

Run:
java SyncDemo

Example Output
5 x 1 = 5
5 x 2 = 10
5 x 3 = 15
5 x 4 = 20
5 x 5 = 25
10 x 1 = 10
10 x 2 = 20
10 x 3 = 30
10 x 4 = 40
10 x 5 = 50





2. Write a Java Program for the implementation of scrollable ResultSet. Assume Teacher
table with attributes (TID, TName, Salary) is already created



CREATE TABLE teacher(tid INT, tname VARCHAR(50),salary INT);

INSERT INTO teacher VALUES (1, 'Amit', 25000);
INSERT INTO teacher VALUES (2, 'Sneha', 30000);
INSERT INTO teacher VALUES (3, 'Raj', 28000);




Save as ScrollableRS.java


import java.sql.*;

public class ScrollableRS {
    public static void main(String[] args) {
        try {
            Class.forName("org.postgresql.Driver");

            Connection con = DriverManager.getConnection(
                    "jdbc:postgresql://localhost:5432/testdb",
                    "postgres",
                    "password"
            );

            Statement st = con.createStatement(
                    ResultSet.TYPE_SCROLL_INSENSITIVE,
                    ResultSet.CONCUR_READ_ONLY
            );

            ResultSet rs = st.executeQuery("SELECT * FROM teacher");

            System.out.println("Forward Direction:");
            while (rs.next()) {
                System.out.println(rs.getInt(1) + " " + rs.getString(2) + " " + rs.getInt(3));
            }

            System.out.println("\nBackward Direction:");
            while (rs.previous()) {
                System.out.println(rs.getInt(1) + " " + rs.getString(2) + " " + rs.getInt(3));
            }

            con.close();
        } catch (Exception e) {
            System.out.println(e);
        }
    }
}








javac -cp ".:postgresql-42.7.3.jar" ScrollableRS.java

java -cp ".:postgresql-42.7.3.jar" ScrollableRS
