.1 Consider a system with ‘m’ processes and ‘n’ resource types. Accept number of
instances for every resource type. For each process accept the allocation and maximum
requirement matrices. Write a program to display the contents of need matrix and to
check if the given request of a process can be granted immediately or not



#include <stdio.h>

#define MAXP 10
#define MAXR 10

int main()
{
    int p, r;
    int allocation[MAXP][MAXR], max[MAXP][MAXR], need[MAXP][MAXR];
    int total[MAXR], allocated[MAXR] = {0}, available[MAXR];
    int request[MAXR], process;
    int i, j, flag = 1;

    printf("Enter number of processes: ");
    scanf("%d", &p);

    printf("Enter number of resource types: ");
    scanf("%d", &r);

    printf("Enter number of instances for each resource:\n");
    for(i = 0; i < r; i++)
    {
        printf("Resource %d: ", i);
        scanf("%d", &total[i]);
    }

    printf("Enter Allocation Matrix:\n");
    for(i = 0; i < p; i++)
    {
        for(j = 0; j < r; j++)
        {
            scanf("%d", &allocation[i][j]);
            allocated[j] += allocation[i][j];
        }
    }

    printf("Enter Maximum Matrix:\n");
    for(i = 0; i < p; i++)
    {
        for(j = 0; j < r; j++)
        {
            scanf("%d", &max[i][j]);
        }
    }

    for(i = 0; i < r; i++)
    {
        available[i] = total[i] - allocated[i];
    }

    for(i = 0; i < p; i++)
    {
        for(j = 0; j < r; j++)
        {
            need[i][j] = max[i][j] - allocation[i][j];
        }
    }

    printf("\nNeed Matrix:\n");
    for(i = 0; i < p; i++)
    {
        printf("P%d: ", i);
        for(j = 0; j < r; j++)
        {
            printf("%d ", need[i][j]);
        }
        printf("\n");
    }

    printf("\nAvailable Resources:\n");
    for(i = 0; i < r; i++)
        printf("%d ", available[i]);
    printf("\n");

    printf("\nEnter process number for request: ");
    scanf("%d", &process);

    printf("Enter request for process P%d:\n", process);
    for(i = 0; i < r; i++)
        scanf("%d", &request[i]);

    for(i = 0; i < r; i++)
    {
        if(request[i] > need[process][i])
        {
            flag = 0;
            printf("\nRequest cannot be granted immediately.\n");
            printf("Reason: Request is greater than Need.\n");
            break;
        }
    }

    if(flag == 1)
    {
        for(i = 0; i < r; i++)
        {
            if(request[i] > available[i])
            {
                flag = 0;
                printf("\nRequest cannot be granted immediately.\n");
                printf("Reason: Resources are not available.\n");
                break;
            }
        }
    }

    if(flag == 1)
    {
        printf("\nRequest can be granted immediately.\n");
    }

    return 0;
}

Q.2 Write a simulation program for disk scheduling using SSTF algorithm. Accept total
number of disk blocks, disk request string, and current head position from the user.
Display the list of request in the order in which it is served. Also display the total number
of head moments.
24, 90, 133, 43, 188, 70, 37, 55
Start Head Position: 58




#include <stdio.h>
#include <stdlib.h>

int main()
{
    int req[100], visited[100];
    int n, head, total = 0;
    int i, j, min, index, diff;

    printf("Enter total number of requests: ");
    scanf("%d", &n);

    printf("Enter request sequence:\n");
    for(i = 0; i < n; i++)
    {
        scanf("%d", &req[i]);
        visited[i] = 0;
    }

    printf("Enter initial head position: ");
    scanf("%d", &head);

    printf("\nOrder of service using SSTF:\n");
    printf("%d ", head);

    for(i = 0; i < n; i++)
    {
        min = 9999;
        index = -1;

        for(j = 0; j < n; j++)
        {
            if(visited[j] == 0)
            {
                diff = abs(req[j] - head);
                if(diff < min)
                {
                    min = diff;
                    index = j;
                }
            }
        }

        visited[index] = 1;
        total += abs(req[index] - head);
        head = req[index];
        printf("-> %d ", head);
    }

    printf("\n\nTotal head movements = %d\n", total);

    return 0;
}
