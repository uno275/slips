Q.1 Write a program to simulate Indexed file allocation method. Assume disk with n
number of blocks. Give value of n as input. Randomly mark some block as allocated and
accordingly maintain the list of free blocks Write menu driver program with menu
options as mentioned above and implement each option.
• Show Bit Vector
• Show Directory
• Delete Already File
• Exit



#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <time.h>

#define MAX 100

struct file
{
    char name[20];
    int indexBlock;
    int blocks[20];
    int count;
};

int bit[MAX];
int n, fileCount = 0;
struct file dir[20];

void initDisk()
{
    int i;
    srand(time(0));
    for(i = 0; i < n; i++)
        bit[i] = rand() % 2;   // 0 = free, 1 = allocated
}

void showBitVector()
{
    int i;
    printf("\nBit Vector:\n");
    for(i = 0; i < n; i++)
        printf("%d ", bit[i]);
    printf("\n");
}

void showDirectory()
{
    int i, j;
    if(fileCount == 0)
    {
        printf("\nDirectory is empty.\n");
        return;
    }

    printf("\nDirectory:\n");
    printf("File Name\tIndex Block\tAllocated Blocks\n");
    for(i = 0; i < fileCount; i++)
    {
        printf("%s\t\t%d\t\t", dir[i].name, dir[i].indexBlock);
        for(j = 0; j < dir[i].count; j++)
        {
            printf("%d", dir[i].blocks[j]);
            if(j != dir[i].count - 1)
                printf(" ");
        }
        printf("\n");
    }
}

void deleteFile()
{
    char fname[20];
    int i, j, found = 0;

    printf("Enter file name to delete: ");
    scanf("%s", fname);

    for(i = 0; i < fileCount; i++)
    {
        if(strcmp(dir[i].name, fname) == 0)
        {
            found = 1;

            bit[dir[i].indexBlock] = 0;

            for(j = 0; j < dir[i].count; j++)
                bit[dir[i].blocks[j]] = 0;

            for(j = i; j < fileCount - 1; j++)
                dir[j] = dir[j + 1];

            fileCount--;
            printf("File deleted successfully.\n");
            break;
        }
    }

    if(!found)
        printf("File not found.\n");
}

int main()
{
    int i, j;

    printf("Enter total number of disk blocks: ");
    scanf("%d", &n);

    initDisk();

    /* sample files to demonstrate delete option */
    for(i = 0; i < n && fileCount < 2; i++)
    {
        if(bit[i] == 0)
        {
            sprintf(dir[fileCount].name, "file%d", fileCount + 1);
            dir[fileCount].indexBlock = i;
            bit[i] = 1;
            dir[fileCount].count = 0;

            for(j = i + 1; j < n && dir[fileCount].count < 2; j++)
            {
                if(bit[j] == 0)
                {
                    dir[fileCount].blocks[dir[fileCount].count++] = j;
                    bit[j] = 1;
                }
            }
            fileCount++;
        }
    }

    int choice;
    do
    {
        printf("\n--- MENU ---\n");
        printf("1. Show Bit Vector\n");
        printf("2. Show Directory\n");
        printf("3. Delete Already File\n");
        printf("4. Exit\n");
        printf("Enter choice: ");
        scanf("%d", &choice);

        switch(choice)
        {
            case 1:
                showBitVector();
                break;
            case 2:
                showDirectory();
                break;
            case 3:
                deleteFile();
                break;
            case 4:
                printf("Exiting...\n");
                break;
            default:
                printf("Invalid choice.\n");
        }
    } while(choice != 4);

    return 0;
}

Q.2 Write a simulation program for disk scheduling using LOOK algorithm. Accept total
number of disk blocks, disk request string, and current head position from the user.
Display the list of request in the order in which it is served. Also display the total number
of head moments.
23, 89, 132, 42, 187, 69, 36, 55
Start Head Position: 40
 Direction: Left




#include <stdio.h>
#include <stdlib.h>

void sort(int arr[], int n)
{
    int i, j, temp;
    for(i = 0; i < n - 1; i++)
    {
        for(j = 0; j < n - i - 1; j++)
        {
            if(arr[j] > arr[j + 1])
            {
                temp = arr[j];
                arr[j] = arr[j + 1];
                arr[j + 1] = temp;
            }
        }
    }
}

int main()
{
    int req[100], left[100], right[100];
    int n, head, i;
    int lcount = 0, rcount = 0, total = 0;

    printf("Enter total number of requests: ");
    scanf("%d", &n);

    printf("Enter request sequence:\n");
    for(i = 0; i < n; i++)
        scanf("%d", &req[i]);

    printf("Enter initial head position: ");
    scanf("%d", &head);

    for(i = 0; i < n; i++)
    {
        if(req[i] < head)
            left[lcount++] = req[i];
        else
            right[rcount++] = req[i];
    }

    sort(left, lcount);
    sort(right, rcount);

    printf("\nOrder of service using LOOK:\n");
    printf("%d ", head);

    for(i = lcount - 1; i >= 0; i--)
    {
        total += abs(head - left[i]);
        head = left[i];
        printf("-> %d ", head);
    }

    for(i = 0; i < rcount; i++)
    {
        total += abs(head - right[i]);
        head = right[i];
        printf("-> %d ", head);
    }

    printf("\n\nTotal head movements = %d\n", total);

    return 0;
}
