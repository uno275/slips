Program 1
Multi-thread: even → square, odd → cube



import java.util.Random;

class NumberThread extends Thread {
    public void run() {
        Random r = new Random();

        try {
            for (int i = 1; i <= 5; i++) {
                int num = r.nextInt(100);
                System.out.println("Generated Number: " + num);

                if (num % 2 == 0) {
                    new SquareThread(num).start();
                } else {
                    new CubeThread(num).start();
                }

                Thread.sleep(1000);
            }
        } catch (Exception e) {
            System.out.println(e);
        }
    }
}

class SquareThread extends Thread {
    int n;

    SquareThread(int n) {
        this.n = n;
    }

    public void run() {
        System.out.println("Square of " + n + " = " + (n * n));
    }
}

class CubeThread extends Thread {
    int n;

    CubeThread(int n) {
        this.n = n;
    }

    public void run() {
        System.out.println("Cube of " + n + " = " + (n * n * n));
    }
}

public class RandomThreadDemo {
    public static void main(String[] args) {
        NumberThread t = new NumberThread();
        t.start();
    }
}




Compile:

javac RandomThreadDemo.java
java RandomThreadDemo

import java.sql.*;

public class ProductDemo {
    public static void main(String[] args) {
        Connection con = null;
        Statement st = null;
        ResultSet rs = null;

        try {
            Class.forName("org.postgresql.Driver");

            con = DriverManager.getConnection(
                "jdbc:postgresql://localhost:5432/postgres",
                "postgres",
                "postgres"
            );

            st = con.createStatement();

            st.executeUpdate("insert into Product values(1,'Pen',10.5)");
            st.executeUpdate("insert into Product values(2,'Book',55.0)");
            st.executeUpdate("insert into Product values(3,'Bag',450.0)");
            st.executeUpdate("insert into Product values(4,'Bottle',80.0)");
            st.executeUpdate("insert into Product values(5,'Box',120.0)");

            rs = st.executeQuery("select * from Product");

            System.out.println("Product Details:");
            while (rs.next()) {
                System.out.println(
                    rs.getInt(1) + " " +
                    rs.getString(2) + " " +
                    rs.getDouble(3)
                );
            }

        } catch (Exception e) {
            System.out.println(e);
        }
    }
}
